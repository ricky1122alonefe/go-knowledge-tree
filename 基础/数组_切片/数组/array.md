## 声明方式
### 先声明再初始化
```
func main() {
	 yinzhengjie [5]string    //声明变量
    str := "a,b,c,d,e"
    list := strings.Split(str,",")
    for k,v := range list{
        yinzhengjie[k] = v  //在上面通过“var”关键字提前声明了变量，现在我们只需要进行初始化就成。注意是没有加“：”哟！
    }
    fmt.Println(yinzhengjie)
}

 var  a [2]int  //先声明变量类型
    a[0] = 100  //对已经声明的
    b := [5]byte{'a','b','c','d','e'} //声明并初始化一个长度为“5”的字节（“uint8”）数组
    c := [3]int{1,22,333}
    d := [...]int{1,2,3,4}  //可以省略长度而采用`...`的方式，Golang会自动根据元素个数来计算长度,
    e := [...]int{10:666}  //不指定长度，对第“11”个元素（下标为10）赋值“666”
    f := [2][3]int{[3]int{1,2,3},[3]int{4,5,6}}
    g := [2][3]int{{100,200,300},{400,500,500}} //如果内部的元素和外部的一样，那么上面的声明可以简化，直接忽略内部的，因为都是“int”型
```
